Introduzione a JavaScript

JavaScript è utilizzato per aggiungere interattività e dinamismo alle pagine web.

Concetti Base di JavaScript

  • Variabili e Tipi di Dati
  • Condizioni e Cicli
  • Funzioni
  • Manipolazione del DOM
  • Eventi
  • Array e Oggetti
  • Async/Await

Esempi Pratici

1. Variabili e Tipi di Dati

// Variabili
let nome = "Alice";
let eta = 30;
let piGreco = 3.14;
console.log(`Nome: ${nome}, Età: ${eta}, Pi Greco: ${piGreco}`);

// Tipi di Dati
let x = 10; // Numero
let y = "Ciao"; // Stringa
let z = true; // Booleano

2. Condizioni e Cicli

// Condizioni
let numero = 10;
if (numero > 5) {
    console.log("Il numero è maggiore di 5");
} else {
    console.log("Il numero è 5 o minore");
}

// Cicli
for (let i = 0; i < 5; i++) {
    console.log(i);
}

3. Funzioni

// Funzioni
function saluta(nome) {
    return `Ciao, ${nome}!`;
}
console.log(saluta("Mondo"));

4. Manipolazione del DOM

// Manipolazione del DOM
document.addEventListener("DOMContentLoaded", () => {
    const titolo = document.querySelector("h1");
    titolo.textContent = "Titolo Cambiato con JavaScript!";
});

5. Eventi

// Eventi
document.addEventListener("DOMContentLoaded", () => {
    const bottone = document.querySelector("button");
    bottone.addEventListener("click", () => {
        alert("Bottone cliccato!");
    });
});

6. Array e Oggetti

// Array
let numeri = [1, 2, 3, 4, 5];
console.log(numeri.length); // Lunghezza dell'array

// Oggetti
let persona = {
    nome: "Alice",
    eta: 30,
    professione: "Sviluppatore"
};
console.log(persona.nome); // Accesso alle proprietà

7. Async/Await

// Async/Await
function eseguiOperazioneAsync() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve("Operazione completata");
        }, 2000);
    });
}

async function eseguiTutto() {
    console.log("Inizio esecuzione");
    const risultato = await eseguiOperazioneAsync();
    console.log(risultato);
    console.log("Fine esecuzione");
}

eseguiTutto();
